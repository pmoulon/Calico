126  template <typename T>
127  static absl::StatusOr<Eigen::Vector3<T>> Project(
128  const Eigen::VectorX<T>& intrinsics,
129  const Eigen::Vector3<T>& omega_sensor_world) {
130  if (intrinsics.size() != kNumberOfParameters) {
131  return absl::InvalidArgumentError(absl::StrCat(
132  "Invalid gyroscope intrinsics size. Expected ", kNumberOfParameters,
133  ", got ", intrinsics.size()));
134  }
135  const T& scale = intrinsics(0);
136  const Eigen::Ref<const Eigen::Vector3<T>> bias =
137  intrinsics.segment(1, 3);
138  return scale * omega_sensor_world + bias;
139  }
